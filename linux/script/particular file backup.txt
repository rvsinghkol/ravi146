

#!/bin/bash
#Purpose = Backup of Important Data

#START
TIME=`date +%b-%d-%y`            # This Command will add date in Backup File Name.
FILENAME=SQLbackup-$TIME.gz    # Here i define Backup file name format.
SRCDIR=/root/SQL                  # Location of Important Data Directory (Source of backup).
DESDIR=/root/Desktop/data                            # Destination of backup file.
tar -cpzf $DESDIR/$FILENAME $SRCDIR 
#END


day=$(date +%F)
Folder="/SQL"

File="/root/Desktop/data/$day.tar.gz"

# Backup Server Configuration
tar cpzf $File $Folder
